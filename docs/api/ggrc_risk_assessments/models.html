RiskAssessment¶

class ggrc_risk_assessments.models.risk_assessment.RiskAssessment¶

Risk Assessment model.

Schema:

CREATE TABLE risk_assessments (
        id INTEGER NOT NULL AUTO_INCREMENT,
        ra_manager_id INTEGER,
        ra_counsel_id INTEGER,
        program_id INTEGER NOT NULL,
        end_date DATE,
        start_date DATE,
        notes TEXT,
        description TEXT,
        title VARCHAR(250) NOT NULL,
        slug VARCHAR(250) NOT NULL,
        updated_at DATETIME NOT NULL,
        modified_by_id INTEGER,
        created_at DATETIME NOT NULL,
        context_id INTEGER,
        PRIMARY KEY (id),
        CONSTRAINT uq_risk_assessments UNIQUE (slug),
        FOREIGN KEY(ra_manager_id) REFERENCES people (id),
        FOREIGN KEY(ra_counsel_id) REFERENCES people (id),
        FOREIGN KEY(program_id) REFERENCES programs (id),
        FOREIGN KEY(context_id) REFERENCES contexts (id)
)
